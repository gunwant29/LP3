import java.util.Random;

public class Randomized_Quick_Sort {

    public static int N = 10;
    public static int[] sequence = new int[N];
    private static final Random RAND = new Random();

    public static void QuickSort(int left, int right) {
        if (right - left <= 0)
            return;
        else {
            int pivotIndex = left + RAND.nextInt(right - left + 1);
            swap(pivotIndex, right);
            int pivot = sequence[right];
            System.out.println("Pivot = " + pivot);

            int partition = partitionIt(left, right, pivot);

            QuickSort(left, partition - 1);
            printSequence(sequence);
            System.out.println();

            QuickSort(partition + 1, right);
            printSequence(sequence);
            System.out.println();
        }
    }

    public static int partitionIt(int left, int right, int pivot) {
        int leftPtr = left - 1;
        int rightPtr = right;

        while (true) {
            while (sequence[++leftPtr] < pivot)
                ;
            while (rightPtr > left && sequence[--rightPtr] > pivot)
                ;

            if (leftPtr >= rightPtr)
                break;
            else
                swap(leftPtr, rightPtr);
        }
        swap(leftPtr, right);
        return leftPtr;
    }

    public static void swap(int dex1, int dex2) {
        int temp = sequence[dex1];
        sequence[dex1] = sequence[dex2];
        sequence[dex2] = temp;
    }

    static void printSequence(int[] arr) {
        for (int i = 0; i < arr.length; i++)
            System.out.print(arr[i] + " ");
    }

    public static void main(String args[]) {
        System.out.println("Sorting of randomly generated numbers using RANDOMIZED QUICK SORT");
        Random random = new Random();

        for (int i = 0; i < N; i++)
            sequence[i] = Math.abs(random.nextInt(100));

        System.out.println("\nOriginal Sequence: ");
        printSequence(sequence);
        System.out.println("\n");

        QuickSort(0, N - 1);

        System.out.println("\nSorted Sequence: ");
        printSequence(sequence);
        System.out.println();
    }
}
